(define (cont-frac D N count)
  (if (= count 0)
      0
      (/ (N count) (+ (D count) (cont-frac D N (- count 1))))))
  
; iterative
(define (iter-cont-frac D N count)
  (define (iter result count)
    (if (= count 0)
        result
        (iter (/ (N count) (+ (D count) result)) (- count 1))))
  (iter 0 count))
